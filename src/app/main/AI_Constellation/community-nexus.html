John Doe Yesterday at 4:30 PM

Ah, you need to specify the event type. Try this:

import React from 'react';

const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
  console.log(e.currentTarget.value);
};

// For input elements
const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
  console.log(e.target.value);
};

I've been working on a cool animation library for React. Here's a snippet of how to use it:

import { useAnimation } from 'react-motion-fx';

function AnimatedComponent() {
  const animation = useAnimation({
    duration: 500,
    easing: 'ease-in-out',
    delay: 100
  });
  
  return (
    <div 
      ref={animation.ref} 
      style={animation.style}
      className="my-animated-element"
    >
      This will animate on mount!
    </div>
  );
}
